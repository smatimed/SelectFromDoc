import tkinter as tk
from tkinter import ttk

root = tk.Tk()
root.geometry("300x200")

# Create a PhotoImage object from an image file
image = tk.PhotoImage(file="path_to_your_image.png")

# Create a ttk.Button with the image
button = ttk.Button(root, text="Click Me", image=image, compound=tk.LEFT)
button.pack(pady=20)

root.mainloop()

checkbutton = ttk.Checkbutton(root, text="Check Me", image=image, compound=tk.LEFT)

import tkinter as tk
from tkinter import ttk

root = tk.Tk()
root.geometry("300x200")

# Create a ttk.Combobox
combo = ttk.Combobox(root, width=15)

# Create a ttk.Style
style = ttk.Style()
style.theme_use('default')

# Create a PhotoImage object for each image
image1 = tk.PhotoImage(file="path_to_image1.png")
image2 = tk.PhotoImage(file="path_to_image2.png")
image3 = tk.PhotoImage(file="path_to_image3.png")

# Add choices to the Combobox
combo['values'] = ('Option 1', 'Option 2', 'Option 3')

# Set the layout for each choice
style.map('TCombobox', fieldbackground=[('readonly', 'white')])
style.map('TCombobox', background=[('readonly', 'white')])
style.map('TCombobox', foreground=[('readonly', 'black')])

# Set the images for each choice
style.map('TCombobox', fieldbackground=[('focus', 'white', image1)])
style.map('TCombobox', fieldbackground=[('pressed', 'white', image2)])
style.map('TCombobox', fieldbackground=[('active', 'white', image3)])

combo.pack(pady=20)
root.mainloop()
